    Kubernetes部署golang服务

    # 前提

    • 需要在docker hub 注册一个账号，用于上传镜像使用

    # 创建web应用

    # web应用代码如下：
    package main

import (
    "fmt"
    "net/http"
)

func index(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "hello world")
}

func main() {
    http.HandleFunc("/", index)
    fmt.Println("server start")
    http.ListenAndServe(":3000", nil)
}

RUN go mod init app

docker push calm2020/go-app-img:latest

    # k8s部署golang服务

    部署yaml文件如下

    apiVersion: apps/v1
kind: Deployment
metadata:
  name: my-go-app
spec:
  replicas: 1
  selector:
    matchLabels:
      app: go-app
  template:
    metadata:
      labels:
        app: go-app
    spec:
      containers:
      - name: go-app-container
        image: calm2020/go-app-img
        resources:
          limits:
            memory: "128Mi"
            cpu: "500m"
        ports:
        - containerPort: 3000
